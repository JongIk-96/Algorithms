function solution(scores) {
    let answer = '';
    let grades = [];
    let grade = ''
    for(let i=0; i<scores.length; i++){
        let sum=0;
        let min=101;
        let max=-1;
        let len = scores.length;
        for(let j=0; j<scores[i].length; j++){
            if(scores[j][i]<=min){
                // if(min == scores[i][j]){       
                // }
                min = scores[j][i]  
            }
            if(scores[j][i]>=max){
                // if(max == scores[i][j]){
                // }
                max = scores[j][i]
            }
            sum+=scores[j][i];
        }
        console.log(max)
        console.log(min)
        for(let j=0; j<scores[i].length; j++){
            if(max !=min ){
                if(i==j && scores[j][i] == min){
                    sum-=scores[j][i]
                    len -= 1
                }
                if(i==j && scores[j][i] == max){
                    sum-=scores[j][i]   
                    len -= 1
                }
            }
        }
        console.log(sum)
        let avg = sum/len
        if(avg>=90){
            grade = 'A'
        }else if(avg<90 && avg>=80){
            grade = 'B'
        }else if(avg<80 && avg>=70){
            grade = 'C'
        }else if(avg<70 && avg>=50){
            grade = 'D'
        }else{
            grade = 'F'
        }
        grades[i] = grade
        console.log(grades)
        answer +=grades[i]
    }
    return answer;
}
